\usepackage{fontspec}
\usepackage{unicode-math}

% Texte principal + maths
\setmainfont{Cambria}
\setmathfont{Cambria Math}
\setsansfont{Calibri}

% Police monospace (blocs de code) avec fallback Unicode large
% Consolas pour le texte courant, Segoe UI Symbol pour les
% caracteres mathematiques manquants (angle brackets, tensor symbols...)
\setmonofont{Consolas}[
  Scale         = 0.9,
  BoldFont      = {Consolas Bold},
  FallbackFonts = {Segoe UI Symbol, Cambria Math}
]
